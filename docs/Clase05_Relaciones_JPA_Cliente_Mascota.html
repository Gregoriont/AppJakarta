Entidad Cliente
DIAPOSITIVA 5 — Cliente.java con @OneToMany
📄 Cliente.java
package com.veterinaria.modelo;

import jakarta.persistence.*;
import jakarta.validation.constraints.*;
import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

@Entity
@Table(name = "cliente")
@NamedQueries({
  @NamedQuery(name = "Cliente.findAll",
    query = "SELECT c FROM Cliente c ORDER BY c.nombre"),
  @NamedQuery(name = "Cliente.findByDocumento",
    query = "SELECT c FROM Cliente c WHERE c.documento = :doc")
})
public class Cliente implements Serializable {

  @Id
  @GeneratedValue(strategy = GenerationType.IDENTITY)
  private Long id;

  @NotBlank(message = "El documento es obligatorio")
  @Size(max = 20)
  @Column(nullable = false, unique = true, length = 20)
  private String documento;

  @NotBlank(message = "El nombre es obligatorio")
  @Size(max = 100)
  @Column(nullable = false, length = 100)
  private String nombre;

  // ═══ @OneToMany: el lado "uno" de la relación ═══════════
  // mappedBy: nombre del campo en Mascota que tiene la FK
  // cascade ALL: guardar/eliminar cliente propaga a sus mascotas
  // orphanRemoval: si quito una mascota de la lista, se elimina en BD
  @OneToMany(
    mappedBy      = "cliente",   // ← campo en Mascota.java
    cascade       = CascadeType.ALL,
    fetch         = FetchType.LAZY,
    orphanRemoval = true
  )
  private List<Mascota> mascotas = new ArrayList<>();
  // NO genera columna en tabla cliente
  // La FK ya está en tabla mascota (cliente_id)

  // ── Getters y Setters ──────────────────────────────────
  public Long   getId()               { return id; }
  public String getDocumento()         { return documento; }
  public void   setDocumento(String d) { this.documento = d; }
  public String getNombre()            { return nombre; }
  public void   setNombre(String n)    { this.nombre = n; }
  public List<Mascota> getMascotas()  { return mascotas; }
  public void setMascotas(List<Mascota> m){ this.mascotas = m; }
}Cliente.java

Entidad Mascota y @ManyToOne
DIAPOSITIVA 6 — Mascota.java con @ManyToOne y @JoinColumn
📄 Mascota.java
package com.veterinaria.modelo;

import jakarta.persistence.*;
import jakarta.validation.constraints.*;
import java.io.Serializable;
import java.time.LocalDate;

@Entity
@Table(name = "mascota")
@NamedQueries({
  @NamedQuery(name = "Mascota.findAll",
    query = "SELECT m FROM Mascota m JOIN FETCH m.cliente ORDER BY m.nombre"),
  @NamedQuery(name = "Mascota.findByCliente",
    query = "SELECT m FROM Mascota m WHERE m.cliente.id = :clienteId ORDER BY m.nombre")
})
public class Mascota implements Serializable {

  @Id
  @GeneratedValue(strategy = GenerationType.IDENTITY)
  private Long id;

  @NotBlank(message = "El nombre es obligatorio")
  @Size(max = 80)
  @Column(nullable = false, length = 80)
  private String nombre;

  @Size(max = 100)
  @Column(length = 100)
  private String raza;

  @Column(columnDefinition = "TEXT")
  private String observaciones;

  @Column
  private LocalDate fecha;  // fecha de nacimiento

  // ═══ @ManyToOne: el lado "muchos" — tiene la FK ══════════
  //
  // @ManyToOne: muchas mascotas → un cliente
  // fetch=LAZY: NO carga cliente hasta que se llame getCliente()
  // optional=false: la mascota DEBE tener un cliente (NOT NULL)
  //
  // @JoinColumn: define el nombre de la columna FK en tabla mascota
  // name="cliente_id" → columna cliente_id en la tabla mascota
  @ManyToOne(fetch = FetchType.LAZY, optional = false)
  @JoinColumn(name = "cliente_id", nullable = false)
  private Cliente cliente;
  // EclipseLink genera: FOREIGN KEY (cliente_id) REFERENCES cliente(id)

  // ── Getters y Setters ──────────────────────────────────
  public Long      getId()                 { return id; }
  public String    getNombre()             { return nombre; }
  public void      setNombre(String n)      { this.nombre = n; }
  public String    getRaza()               { return raza; }
  public void      setRaza(String r)        { this.raza = r; }
  public String    getObservaciones()      { return observaciones; }
  public void      setObservaciones(String o){ this.observaciones = o; }
  public LocalDate getFecha()              { return fecha; }
  public void      setFecha(LocalDate f)   { this.fecha = f; }
  public Cliente   getCliente()            { return cliente; }
  public void      setCliente(Cliente c)   { this.cliente = c; }

  // Helper para mostrar en la vista
  public String getNombreConDueno() {
    return nombre + " (Dueño: " + cliente.getNombre() + ")";
  }
}Mascota.java

@OneToMany y mappedBy
DIAPOSITIVA 8 — ¿Qué es mappedBy?
📌
mappedBy le dice a JPA: "La FK no está en mi tabla. El campo que la gestiona está en la otra clase, y se llama exactamente como dice mappedBy."
En nuestro caso: mappedBy = "cliente" porque en Mascota.java el campo se llama cliente.
❌ Sin mappedBy — ERROR
// ❌ Esto crea una SEGUNDA tabla intermedia
// llamada "cliente_mascotas" — no es lo que queremos
@OneToMany   // sin mappedBy
private List<Mascota> mascotas;

// JPA genera:
// CREATE TABLE cliente_mascotas (
//   cliente_id BIGINT,
//   mascotas_id BIGINT
// )   ← tabla extra no deseada
✅ Con mappedBy — CORRECTO
// ✅ Le dice a JPA que la FK ya está en mascota
// No crea ninguna columna extra en tabla cliente
@OneToMany(mappedBy = "cliente")
private List<Mascota> mascotas;

// "cliente" = nombre exacto del campo
// en Mascota.java que tiene @ManyToOne
//
// JPA no genera nada nuevo en tabla cliente
// La FK cliente_id ya existe en tabla mascota ✅
DIAPOSITIVA 9 — Resumen: ¿quién tiene qué?
Clase JavaAnotaciónTabla MySQLGenera
Mascota @ManyToOne + @JoinColumn(name="cliente_id") mascota Columna cliente_id (FK) en tabla mascota
Cliente @OneToMany(mappedBy="cliente") cliente Nada — solo permite navegar cliente.getMascotas()

FetchType y CascadeType
DIAPOSITIVA 10 — LAZY vs EAGER
✅ LAZY — carga bajo demanda
@ManyToOne(fetch = FetchType.LAZY)
private Cliente cliente;

// Paso 1: solo carga la mascota
Mascota m = em.find(Mascota.class, 1L);
// → SELECT id, nombre, raza FROM mascota WHERE id=1
//   (cliente NO se carga todavía)

// Paso 2: al acceder al cliente → carga ahora
String nombreDueno = m.getCliente().getNombre();
// → SELECT id, documento, nombre FROM cliente WHERE id=?
⚠️ EAGER — carga siempre con JOIN
@ManyToOne(fetch = FetchType.EAGER)
private Cliente cliente;

// Carga mascota Y cliente juntos, siempre
Mascota m = em.find(Mascota.class, 1L);
// → SELECT m.id, m.nombre, m.raza,
//          c.id, c.documento, c.nombre
//   FROM mascota m
//   LEFT JOIN cliente c ON m.cliente_id = c.id
//   WHERE m.id = 1
//
// Lo uses o no, siempre carga el cliente ⚠️
DIAPOSITIVA 11 — CascadeType en @OneToMany
ValorQué propagaEjemplo práctico
PERSISTem.persist()Al guardar un Cliente nuevo, guarda también sus mascotas nuevas
MERGEem.merge()Al actualizar un Cliente, actualiza sus mascotas
REMOVEem.remove()Al eliminar un Cliente, elimina todas sus mascotas
ALLTodas las anterioresTodo se propaga del padre al hijo
⚠️
Cuidado con CascadeType.REMOVE: Si una mascota tiene turnos, eliminar el cliente en cascada intentará eliminar las mascotas, pero fallará si los turnos tienen FK restrictiva. En sistemas reales preferir baja lógica (campo activo=false) en lugar de DELETE físico.

DAO y JPQL con relaciones
DIAPOSITIVA 12 — MascotaDAO.java
📄 MascotaDAO.java
package com.veterinaria.dao;
import com.veterinaria.modelo.Mascota;
import jakarta.enterprise.context.ApplicationScoped;
import jakarta.transaction.Transactional;
import java.util.List;

@ApplicationScoped
public class MascotaDAO extends GenericDAO<Mascota, Long> {

  @Override
  protected Class<Mascota> getEntityClass() { return Mascota.class; }

  // JOIN FETCH evita el problema N+1:
  // trae mascota + cliente en 1 sola consulta SQL
  public List<Mascota> listarTodas() {
    return em.createNamedQuery("Mascota.findAll", Mascota.class)
             .getResultList();
  }

  // Mascotas de un cliente específico
  public List<Mascota> listarPorCliente(Long clienteId) {
    return em.createNamedQuery("Mascota.findByCliente", Mascota.class)
             .setParameter("clienteId", clienteId)
             .getResultList();
  }
}MascotaDAO.java

Servicio: guardar Mascota con su Cliente
DIAPOSITIVA 14 — El problema: el formulario envía un ID, no un objeto
⚠️
El <h:selectOneMenu> del formulario devuelve el ID del cliente (un Long). Pero Mascota.setCliente() espera un objeto Cliente. El Servicio es quien resuelve esta diferencia consultando la BD con el ID.
📄 MascotaServicio.java
@ApplicationScoped
public class MascotaServicio {

  @Inject private MascotaDAO  mascotaDAO;
  @Inject private ClienteDAO  clienteDAO;

  public List<Mascota> listarTodas()  { return mascotaDAO.listarTodas(); }
  public Mascota buscarPorId(Long id) { return mascotaDAO.buscarPorId(id); }

  @Transactional
  public void guardar(Mascota mascota, Long clienteId) {

    // Validar que llegó un clienteId
    if (clienteId == null)
      throw new IllegalArgumentException("Debe seleccionar un dueño.");

    // Buscar el objeto Cliente en la BD usando el ID
    Cliente cliente = clienteDAO.buscarPorId(clienteId);
    if (cliente == null)
      throw new IllegalArgumentException("El cliente seleccionado no existe.");

    // Setear el objeto Cliente en la entidad Mascota
    mascota.setCliente(cliente);  // ← clave: setear el objeto, no el ID

    if (mascota.getId() == null)
      mascotaDAO.guardar(mascota);    // INSERT
    else
      mascotaDAO.actualizar(mascota); // UPDATE
  }

  @Transactional
  public void eliminar(Long id) { mascotaDAO.eliminar(id); }
}MascotaServicio.java

@Named("mascotaBean") @ViewScoped
public class MascotaBean implements Serializable {

  @Inject private MascotaServicio mascotaServicio;
  @Inject private ClienteServicio  clienteServicio;

  private List<Mascota> mascotas;
  private List<Cliente> clientes;     // para el selector
  private Mascota        mascotaSeleccionada;
  private Long            clienteIdSeleccionado; // ID del selector
  private boolean         modoEdicion = false;

  @PostConstruct
  public void init() {
    mascotaSeleccionada   = new Mascota();
    clientes              = clienteServicio.listarTodos();
    cargarMascotas();
  }

  public void cargarMascotas() {
    mascotas = mascotaServicio.listarTodas();
  }

  public void prepararEdicion(Mascota m) {
    mascotaSeleccionada   = mascotaServicio.buscarPorId(m.getId());
    clienteIdSeleccionado = mascotaSeleccionada.getCliente().getId();
    modoEdicion           = true;
  }

  public void guardar() {
    try {
      // Pasa el ID del selector al Servicio
      mascotaServicio.guardar(mascotaSeleccionada, clienteIdSeleccionado);
      cargarMascotas();
      mascotaSeleccionada   = new Mascota();
      clienteIdSeleccionado = null;
      modoEdicion           = false;
      addMsg(SEVERITY_INFO, "Éxito", "Mascota guardada.");
    } catch (IllegalArgumentException e) {
      addMsg(SEVERITY_WARN, "Atención", e.getMessage());
    } catch (Exception e) {
      addMsg(SEVERITY_ERROR, "Error", "No se pudo guardar.");
    }
  }

  public void eliminar(Mascota m) {
    mascotaServicio.eliminar(m.getId());
    cargarMascotas();
    addMsg(SEVERITY_INFO, "Eliminada", m.getNombre() + " eliminada.");
  }

  public void cancelar() {
    mascotaSeleccionada   = new Mascota();
    clienteIdSeleccionado = null;
    modoEdicion           = false;
    cargarMascotas();
  }

  private void addMsg(FacesMessage.Severity s, String r, String d) {
    FacesContext.getCurrentInstance()
        .addMessage(null, new FacesMessage(s, r, d));
  }

  // Getters y Setters
  public List<Mascota> getMascotas()         { return mascotas; }
  public List<Cliente> getClientes()          { return clientes; }
  public Mascota getMascotaSeleccionada()      { return mascotaSeleccionada; }
  public void setMascotaSeleccionada(Mascota m){ mascotaSeleccionada = m; }
  public Long getClienteIdSeleccionado()       { return clienteIdSeleccionado; }
  public void setClienteIdSeleccionado(Long id){ clienteIdSeleccionado = id; }
  public boolean isModoEdicion()               { return modoEdicion; }
}MascotaBean.java

DIAPOSITIVA 17 — Por qué necesitamos <f:converter>
❌ Sin f:converter — ConverterException

El <h:selectOneMenu> envía el valor como String (HTML solo maneja texto). El Bean espera un Long. Sin converter → ConverterException: Cannot convert 'null' of type class java.lang.String

✅ Con f:converter — correcto

El converter "jakarta.faces.Long" convierte automáticamente el String "3" al Long 3L antes de setear el Bean. Es obligatorio en selectores que usan IDs numéricos.

<f:converter converterId="jakarta.faces.Long"/>
// "3" (String HTML) → 3L (Long Java)
✅
